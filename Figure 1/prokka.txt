#!/bin/sh
source /opt/asn/etc/asn-bash-profiles-special/modules.sh
module load anaconda/3-2020.02
for file in *.fna; do tag=${file%.fna}; \
prokka --kingdom Bacteria --outdir prokka_"$tag" --force --addgenes --prefix "$tag" --locustag "$tag" --increment 10 \
 --genus Xylella --gcode 11 "$file"; done
